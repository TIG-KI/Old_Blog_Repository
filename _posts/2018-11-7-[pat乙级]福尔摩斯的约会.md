## [pat乙级] 福尔摩斯的约会
```C++
#include<stdio.h>
#include<string.h>

typedef  char string[60]; //定义字符串

void P1(string a,string b, int &i, int &tmin) //输出星期 &表引用，传进来的i值将对会循环的进程进行记录
{
    int l1 = strlen(a);
    int l2 = strlen(b);
    tmin = l1 < l2? l1 : l2;
    for(i = 0; i < tmin; i++)
    {
        if(a[i] == b[i] && (a[i] >= 'A' && a[i]<='G'))
        {
            switch(a[i])
            {
            case 'A':
                printf("MON ");
                break;
            case 'B':
                printf("TUE ");
                break;
            case 'C':
                printf("WED ");
                break;
            case 'D':
                printf("THU ");
                break;
            case 'E':
                printf("FRI ");
                break;
            case 'F':
                printf("SAT ");
                break;
            case 'G':
                printf("SUN ");
                break;
            }
            break;
        }
    }
}

void P2(string a,string b, int i,int tmin) //输出时间的小时数
{

    for(i=i+1; i < tmin; i++)
    {
        if(a[i] == b[i])
        {
            if(a[i] >= 'A' && a[i] <= 'N')
            {
                printf("%d : ",a[i]-55);
                break;
            }
            if(a[i] >= '0' && a[i] <= '9')
            {
                printf("0%d : ",a[i]-48);
                break;
            }
        }
    }
}

void P3(string c, string d) //输出时间的分钟数
{
    int tmin2 = strlen(c) < strlen(d) ? strlen(c):strlen(d);
    for(int j = 0; j < tmin2; j++)
    {
        if(c[j] == d[j] && ((c[j] > 'a' && c[j] < 'z') || (c[j] > 'A' && c[j] < 'Z')))
        {
            if(j < 9)
            {
                printf("0%d",j);
            }
            else
            {
                printf("%d",j);
            }
            break;
        }
    }
}

int main()
{
    string a, b, c, d;
    int i, tmin; //分别用以记录 循环的状态 和记录两个字符串中长度最短的那个字符串的长度
    
    gets(a);
    gets(b);
    gets(c);
    gets(d);
    
    P1(a, b, i, tmin);
    P2(a, b, i, tmin);
    P3(c, d);

    return 0;
}
```
